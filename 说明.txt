    'default': {
        'ENGINE': 'django.db.backends.mysql',
        'NAME': 'agriculture_knowledgegraph',  # 数据库名
        'USER': 'myuser',  # 数据库用户名
        'PASSWORD': 'xy639a58',  # 数据库密码
        'HOST': '127.0.0.1',  # 数据库主机名或IP地址
        'PORT': '3306',  # 数据库端口号（默认为3306）
    }
}
